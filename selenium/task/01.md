from selenium import webdriver

# 导入浏览器驱动路径
executable_path = r"d:\tools\webdrivers\chromedriver.exe"


# 指定是chrome 的驱动

driver = webdriver.Chrome(executable_path)

# get 方法 打开指定网址
driver.get('http://www.weather.com.cn/html/province/jiangsu.shtml')

# todo
